% --- Beamer Configuration ---
\metroset{block=fill}

% Frame title bar and standout frames
\setbeamercolor{palette primary}{bg=phluxdark, fg=white}

\setbeamercolor{progress bar}{fg=phluxaccent}
\setbeamercolor{title separator}{fg=phluxaccent}
\setbeamercolor{alerted text}{fg=phluxaccent}
\renewcommand{\arraystretch}{1.2}

% --- Dark Title & Section Slides ---
% Title slide: created at \begin{document} via \AtBeginDocument.
% Template-body approaches fail because pandoc/Metropolis resets colors
% after \begin{document}. Preamble-defined hooks run in the right scope.
\AtBeginDocument{%
  \begingroup
    \setbeamercolor{background canvas}{bg=phluxdark}%
    \setbeamercolor{normal text}{fg=white}%
    \setbeamercolor{title}{fg=white}%
    \setbeamercolor{subtitle}{fg=white!80}%
    \setbeamercolor{author}{fg=white!80}%
    \setbeamercolor{date}{fg=white!80}%
    \setbeamercolor{institute}{fg=white!80}%
    \setbeamercolor{title separator}{fg=phluxaccent}%
    \usebeamercolor[fg]{normal text}%
    \begin{frame}[plain,noframenumbering]
      \titlepage
    \end{frame}
  \endgroup
}
% Section dividers: Metropolis overrides any template-level \AtBeginSection,
% so we override it here in the preamble (loaded after the theme).
\AtBeginSection{%
  \begingroup
    \setbeamercolor{background canvas}{bg=phluxdark}%
    \setbeamercolor{normal text}{fg=white}%
    \setbeamercolor{progress bar}{fg=phluxaccent}%
    \usebeamercolor[fg]{normal text}%
    \begin{frame}[plain,noframenumbering]
      \sectionpage
    \end{frame}
  \endgroup
}
